OneDeck - 生成卡片 Pop Up 动画逻辑整理

整理范围：逻辑层捕获 → 动画数据结构 → 播放器调度 → 视觉层实现

1. 核心设计思想

生成（或需要引起注意的）卡片不会立即静默进入牌堆，而是执行一段 Pop Up → (中间效果) → Slot In 的三段式动画，让玩家清晰感知到新卡片的加入。

  • Pop Up：卡片从当前位置向上浮起、放大、前移，进入玩家视野。
  • 中间效果（可选）：投射物飞行、状态效果变更、放逐销毁等。
  • Slot In：卡片从浮起位置平滑落回牌堆的正确索引位置。

3. 逻辑层捕获 AnimationRequest

3.1 新增临时卡片 — AddTempCard

文件：Assets/Scripts/Effects/AddTempCard.cs

private void CapturePopUpSlotInForNewCard(GameObject newCard)
{
    var recorder = EffectChainManager.Me?.currentEffectRecorder?.GetComponent<EffectRecorder>();
    if (recorder != null)
    {
        recorder.animationRequests.Add(new AnimationRequest
        {
            type = AnimationRequestType.PopUp,
            targetCard = newCard
        });
        recorder.animationRequests.Add(new AnimationRequest
        {
            type = AnimationRequestType.SlotIn,
            targetCard = newCard
        });
    }
}

public void ApplyPowerToCardWithProjectile(CardScript targetCard, int amount)
{
    ApplyPowerToCardInternal(targetCard, amount); // 逻辑立即执行

    var recorder = EffectChainManager.Me?.currentEffectRecorder?.GetComponent<EffectRecorder>();
    if (recorder != null)
    {
        // 1. Pop Up 目标卡
        recorder.animationRequests.Add(new AnimationRequest
        {
            type = AnimationRequestType.PopUp,
            targetCard = targetCard.gameObject
        });
        // 2. 投射物飞行（从来源卡到目标卡）
        recorder.animationRequests.Add(new AnimationRequest
        {
            type = AnimationRequestType.StatusEffectProjectile,
            attackerCard = myCard,
            targetCard = targetCard.gameObject
        });
        // 3. Slot In 落回牌堆
        recorder.animationRequests.Add(new AnimationRequest
        {
            type = AnimationRequestType.SlotIn,
            targetCard = targetCard.gameObject
        });
    }
}

3.3 消耗状态效果 — ConsumeStatusEffect

文件：Assets/Scripts/Effects/EffectScript.cs — 基类提供的通用方法

protected void CapturePopUpStatusEffectChangeSlotIn(
    GameObject targetCard, EnumStorage.StatusEffect effect, int amount)
{
    var recorder = EffectChainManager.Me?.currentEffectRecorder?.GetComponent<EffectRecorder>();
    if (recorder == null) return;

    // 1. Pop Up
    recorder.animationRequests.Add(new AnimationRequest
    {
        type = AnimationRequestType.PopUp, targetCard = targetCard
    });
    // 2. 状态效果变更（染色 + 粒子）
    recorder.animationRequests.Add(new AnimationRequest
    {
        type = AnimationRequestType.StatusEffectChange,
        targetCard = targetCard,
        statusEffect = effect,
        statusEffectAmount = amount
    });
    // 3. Slot In
    recorder.animationRequests.Add(new AnimationRequest
    {
        type = AnimationRequestType.SlotIn, targetCard = targetCard
    });
}

被 ConsumeStatusEffect.ConsumeOwnStatusEffect() 与 ConsumeRandomEnemyCardsStatusEffect() 调用。

5. 动画播放器调度

5.1 播放前恢复 Deck Focus

如果牌堆当前处于 Peel/Focus 状态，PopUp、SlotIn 等位置相关动画会先强制恢复普通牌堆布局：

if (request.type == AnimationRequestType.MoveToBottomBatch ||
    request.type == AnimationRequestType.MoveToTopBatch ||
    request.type == AnimationRequestType.PopUp ||
    request.type == AnimationRequestType.SlotIn)
{
    var combatUX = visuals as CombatUXManager;
    if (combatUX != null && combatUX.IsDeckFocused)
    {
        yield return combatUX.StartCoroutine(combatUX.RestoreDeckFocusCoroutine());
    }
}

6.2 PopUpCard

public void PopUpCard(GameObject logicalCard, Action onComplete = null)
{
    // 1. 查找物理卡片
    var physicalCard = GetPhysicalCardFromLogicalCard(cardScript);
    var physScript = physicalCard.GetComponent<CardPhysObjScript>();

    // 2. 清除现有补间，防止冲突
    physScript.KillTweens();

    // 3. 计算峰值位置：当前位置 + 上移 + Z前移
    Vector3 currentPos = physicalCard.transform.position;
    Vector3 peakPos = currentPos + Vector3.up * popUpYOffset;
    peakPos.z += popUpZBoost;
    Vector3 peakScale = physicalCardDeckSize * popUpScaleMultiplier;

    // 4. 标记特殊动画中，阻断输入
    physScript.isPlayingSpecialAnimation = true;
    BlockInput(this);

    // 5. DOTween 同时执行位移与缩放
    Sequence seq = DOTween.Sequence();
    seq.Append(physicalCard.transform.DOMove(peakPos, popUpDuration).SetEase(popUpEase));
    seq.Join(physicalCard.transform.DOScale(peakScale, popUpDuration).SetEase(popUpEase));
    seq.OnComplete(() =>
    {
        UnblockInput(this);
        onComplete?.Invoke();
    });
    seq.Play();
}
卡片在 PopUp 完成后会停留在 peakPos，直到 SlotIn 被调用。期间 isPlayingSpecialAnimation = true 会阻止常规牌堆位置更新将其拉回。

6.3 SlotInCard

public void SlotInCard(GameObject logicalCard, Action onComplete = null)
{
    // 1. 查找物理卡片
    var physicalCard = GetPhysicalCardFromLogicalCard(cardScript);
    var physScript = physicalCard.GetComponent<CardPhysObjScript>();

    // 2. 获取当前在 physicalCardsInDeck 中的索引
    int deckIndex = physicalCardsInDeck.IndexOf(physicalCard);
    if (deckIndex < 0)
    {
        // 不在牌堆中（如 revealZone），直接释放标记
        physScript.isPlayingSpecialAnimation = false;
        onComplete?.Invoke();
        return;
    }

    // 3. 计算该索引对应的牌堆目标位置
    Vector3 targetPos = CalculatePositionAtIndex(deckIndex);

    // 4. DOTween 落回 + 恢复缩放
    Sequence seq = DOTween.Sequence();
    seq.Append(physicalCard.transform.DOMove(targetPos, slotInDuration).SetEase(slotInEase));
    seq.Join(physicalCard.transform.DOScale(physicalCardDeckSize, slotInDuration).SetEase(slotInEase));
    seq.OnComplete(() =>
    {
        physScript.isPlayingSpecialAnimation = false;
        physScript.SetTargetPosition(targetPos);
        physScript.SetTargetScale(physicalCardDeckSize);
        UnblockInput(this);
        onComplete?.Invoke();
    });
    seq.Play();
}

9. 关键注意事项

  • isPlayingSpecialAnimation：PopUp 期间设为 true，阻止 UpdateAllPhysicalCardTargets() 把卡片拉回旧位置。
  • Deck Focus 恢复：若牌堆处于 Peel 聚焦状态，PopUp / SlotIn 播放前会自动调用 RestoreDeckFocusCoroutine()，确保动画在正确布局下执行。
  • 输入阻断：PopUp 与 SlotIn 各自通过 BlockInput/UnblockInput 引用计数阻断玩家点击，防止动画中途被打断。
  • 不在牌堆中：SlotIn 时若 physicalCardsInDeck.IndexOf(physicalCard) < 0（如 revealZone），直接释放 isPlayingSpecialAnimation，不做位移动画。
  • 效果链内跳过自动补间：AddPhysicalCardToDeck 检测到 EffectChainManager.Me.currentEffectRecorder != null 时，不会给新卡设置目标位置，完全交由 Recorder 控制。
